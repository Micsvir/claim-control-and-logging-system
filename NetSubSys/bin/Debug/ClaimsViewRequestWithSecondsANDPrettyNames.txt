SELECT     
	dbo.Claims.ClaimID AS 'ID', 
	dbo.Claims.ClaimReceivedDate AS 'Дата', 
	dbo.Claims.ClaimReceivedTime AS 'Время',
	dbo.Claims.ClaimSenderUserName AS 'Имя пользователя', 
	dbo.Claims.ClaimSenderHostName AS 'ПК', 
	dbo.Claims.ClaimSenderHostIP AS 'IP', 
	dbo.Claims.ClaimSenderRoom AS 'Комната', 
        dbo.Claims.TypeOfIssue AS 'Заявка', 
	dbo.Claims.ClaimDiscription AS 'Доп. инфо',	 
	dbo.Claims.ClaimStatus AS 'Статус', 
        dbo.Groups.GroupName AS 'Ответственная группа', 
        GroupOrderer.PersLastName + ' ' + GroupOrderer.PersFirstName + ' ' + GroupOrderer.PersPatronymic AS 'Контролер', 
        dbo.Claims.ExecGroupOrderReceivedDate AS 'Дата назначения группы', 
	dbo.Claims.ExecGroupOrderReceivedTime AS 'Время назначения группы', 
	DATEDIFF(second, CONVERT(datetime, dbo.Claims.ClaimReceivedDate + ' ' + dbo.Claims.ClaimReceivedTime, 0), CONVERT(datetime, dbo.Claims.ExecGroupOrderReceivedDate + ' ' + dbo.Claims.ExecGroupOrderReceivedTime, 0)) AS 'Прошло времени с поступления заявки', 
        Executor.PersLastName + ' ' + Executor.PersFirstName + ' ' + Executor.PersPatronymic AS 'Исполнитель', 
        ExecOrderer.PersLastName + ' ' + ExecOrderer.PersFirstName + ' ' + ExecOrderer.PersPatronymic AS 'Назначивший исполнителя', 
	dbo.Claims.ExecOrderReceivedDate AS 'Дата назначения исполнителя', 
	dbo.Claims.ExecOrderReceivedTime AS 'Время назначения исполнителя', 
	DATEDIFF(second, CONVERT(datetime, dbo.Claims.ExecGroupOrderReceivedDate + ' ' + dbo.Claims.ExecGroupOrderReceivedTime, 0), CONVERT(datetime, dbo.Claims.ExecOrderReceivedDate + ' ' + dbo.Claims.ExecOrderReceivedTime, 0)) AS 'Прошло времени с назначения группы', 
	dbo.Claims.ClaimExecStartReceivedDate AS 'Дата начала выполнения заявки', 
	dbo.Claims.ClaimExecStartReceivedTime AS 'Время начала выполнения заявки', 
	DATEDIFF(second, CONVERT(datetime, dbo.Claims.ExecOrderReceivedDate + ' ' + dbo.Claims.ExecOrderReceivedTime, 0), CONVERT(datetime, dbo.Claims.ClaimExecStartReceivedDate + ' ' + dbo.Claims.ClaimExecStartReceivedTime, 0)) AS 'Прошло времени с назначения исполнителя',  
	dbo.Claims.ClaimExecCompleteReceivedDate AS 'Дата завершения выполнения заявки', 
	dbo.Claims.ClaimExecCompleteReceivedTime AS 'Время завершения выполнения заявки', 
	DATEDIFF(second, CONVERT(datetime, dbo.Claims.ClaimExecStartReceivedDate + ' ' + dbo.Claims.ClaimExecStartReceivedTime), CONVERT(datetime, dbo.Claims.ClaimExecCompleteReceivedDate + ' ' + dbo.Claims.ClaimExecCompleteReceivedTime, 0)) AS 'Продолжительность выполнения заявки',
	dbo.Claims.ClaimExecutionDescription AS 'Предпринятые действия', 
        dbo.CommonCategories.CommonCatName AS 'Категория заявки', 
	dbo.InternalCategories.IntCatName AS 'Внутренняя категория заявки', 
	dbo.Claims.AffectedObjectsCount AS 'Кол-во затронутых объектов'
FROM         
	dbo.Claims LEFT OUTER JOIN
        dbo.Groups ON dbo.Claims.ExecGroupID = dbo.Groups.GroupID LEFT OUTER JOIN
        dbo.Personal AS GroupOrderer ON dbo.Claims.ExecGroupOrdererID = GroupOrderer.PersID LEFT OUTER JOIN
        dbo.Personal AS Executor ON dbo.Claims.ExecID = Executor.PersID LEFT OUTER JOIN
        dbo.Personal AS ExecOrderer ON dbo.Claims.ExecOrdererID = ExecOrderer.PersID LEFT OUTER JOIN
        dbo.CommonCategories ON dbo.Claims.ClaimCommonCategoryID = dbo.CommonCategories.CommonCatID LEFT OUTER JOIN
        dbo.InternalCategories ON dbo.Claims.ClaimInternalCategoryID = dbo.InternalCategories.IntCatID
